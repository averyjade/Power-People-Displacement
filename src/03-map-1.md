function createFloodSystem(level = 0.3) {

  const canvas =
    document.createElement("canvas");

  const ctx =
    canvas.getContext("2d");

  let visible = true;

  canvas.dataset.flood =
    "true";

  Object.assign(
    canvas.style,
    {
      position: "fixed",
      inset: "0",
      width: "100vw",
      height: "100vh",
      pointerEvents: "none",
      zIndex: "5"
    }
  );

  document.body.appendChild(
    canvas
  );

  canvas.width =
    window.innerWidth *
    devicePixelRatio;

  canvas.height =
    window.innerHeight *
    devicePixelRatio;

  ctx.scale(
    devicePixelRatio,
    devicePixelRatio
  );

  function draw() {

    ctx.clearRect(
      0,
      0,
      window.innerWidth,
      window.innerHeight
    );

    if (!visible) {
      requestAnimationFrame(draw);
      return;
    }

    const wave =
      Math.sin(
        performance.now() *
        0.001
      ) * 12;

    const y =
      window.innerHeight *
      (1 - level) +
      wave;

    ctx.fillStyle =
      "rgba(10,136,207,.35)";

    ctx.beginPath();

    ctx.moveTo(
      0,
      window.innerHeight
    );

    ctx.lineTo(
      0,
      y
    );

    ctx.lineTo(
      window.innerWidth,
      y
    );

    ctx.lineTo(
      window.innerWidth,
      window.innerHeight
    );

    ctx.closePath();

    ctx.fill();

    requestAnimationFrame(draw);
  }

  draw();

  // 🌊 CLICK TO REMOVE FLOOD
  window.addEventListener(
    "click",
    () => {

      visible = false;

      canvas.style.display =
        "none";

    }
  );

}

<script type="module">

import maplibregl from "https://esm.sh/maplibre-gl";

const terrainSlider =
document.getElementById("terrainSlider");

const waterSlider =
document.getElementById("waterSlider");

const map =
new maplibregl.Map({

container:"map",

center:[18.0686,59.3293],

zoom:2,

style:
"https://basemaps.cartocdn.com/gl/voyager-gl-style/style.json",

scrollZoom:true

});

map.on("load",()=>{

// -------------------------
// BASE TERRAIN LAYER
// -------------------------

map.addSource("historical-map",{
type:"raster",
tiles:[
"https://gsk.enchart.me/{z}/{x}/{y}.png"
],
tileSize:256,
minzoom:0,
maxzoom:15
});

map.addLayer({

id:"historical-map",

type:"raster",

source:"historical-map",

paint:{
"raster-opacity":0.65
}

});

// terrain slider = fades context
terrainSlider.addEventListener("input",()=>{

map.setPaintProperty(
"historical-map",
"raster-opacity",
0.65 * Number(terrainSlider.value)/100
);

});

// -------------------------
// FLOOD POLYGON
// -------------------------

map.addSource("modern_shape",{
type:"vector",
tiles:[
"https://lule.enchart.me/api/v1/tiles/modern_shape/{z}/{x}/{y}"
]
});

map.addLayer({

id:"flood",

type:"fill",

source:"modern_shape",

"source-layer":"modern_shape",

paint:{
"fill-color":"#4E8DFF",
"fill-opacity":0
}

});

// water slider = rising flood level
waterSlider.addEventListener("input",()=>{

const v = Number(waterSlider.value)/100;

const level = Math.pow(v, 2.2);

map.setPaintProperty(
"flood",
"fill-opacity",
level
);

});

// -------------------------
// CAMERA
// -------------------------

map.fitBounds(
[16.42,67.12,19.35,68.92],
{
padding:40,
maxZoom:15
}
);

});

</script>
