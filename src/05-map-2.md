function createFloodSystem(level = 0.3) {

  const canvas =
    document.createElement("canvas");

  const ctx =
    canvas.getContext("2d");

  let visible = true;

  canvas.dataset.flood =
    "true";

  Object.assign(
    canvas.style,
    {
      position: "fixed",
      inset: "0",
      width: "100vw",
      height: "100vh",
      pointerEvents: "none",
      zIndex: "5"
    }
  );

  document.body.appendChild(
    canvas
  );

  canvas.width =
    window.innerWidth *
    devicePixelRatio;

  canvas.height =
    window.innerHeight *
    devicePixelRatio;

  ctx.scale(
    devicePixelRatio,
    devicePixelRatio
  );

  function draw() {

    ctx.clearRect(
      0,
      0,
      window.innerWidth,
      window.innerHeight
    );

    if (!visible) {
      requestAnimationFrame(draw);
      return;
    }

    const wave =
      Math.sin(
        performance.now() *
        0.001
      ) * 12;

    const y =
      window.innerHeight *
      (1 - level) +
      wave;

    ctx.fillStyle =
      "rgba(10,136,207,.35)";

    ctx.beginPath();

    ctx.moveTo(
      0,
      window.innerHeight
    );

    ctx.lineTo(
      0,
      y
    );

    ctx.lineTo(
      window.innerWidth,
      y
    );

    ctx.lineTo(
      window.innerWidth,
      window.innerHeight
    );

    ctx.closePath();

    ctx.fill();

    requestAnimationFrame(draw);
  }

  draw();

  // 🌊 CLICK TO REMOVE FLOOD
  window.addEventListener(
    "click",
    () => {

      visible = false;

      canvas.style.display =
        "none";

    }
  );

}

import maplibregl from "https://esm.sh/maplibre-gl";

const terrain = document.getElementById("terrain");
const flood = document.getElementById("flood");

const water_color = "#0A88CF";
const town_color = "#287A12";

const map = new maplibregl.Map({
  container: "map",
  zoom: 9.8,
  center: [20.33387, 66.6872],
  style: "https://basemaps.cartocdn.com/gl/voyager-gl-style/style.json",
  scrollZoom: true
});

map.on("load", () => {

  // ─────────────────────────────
  // BASE RASTER
  // ─────────────────────────────
  map.addSource("historical-map", {
    type: "raster",
    tiles: ["https://gsk.enchart.me/{z}/{x}/{y}.png"],
    tileSize: 256
  });

  map.addLayer({
    id: "historical-map",
    type: "raster",
    source: "historical-map",
    paint: {
      "raster-opacity": 0.7
    }
  });

  // ─────────────────────────────
  // MESSAURE POLYGON
  // ─────────────────────────────
  map.addSource("messaure", {
    type: "vector",
    tiles: ["https://lule.enchart.me/api/v1/tiles/messaure/{z}/{x}/{y}"]
  });

  map.addLayer({
    id: "messaure_fill",
    type: "fill",
    source: "messaure",
    "source-layer": "messaure",
    paint: {
      "fill-color": "#0A88CF",
      "fill-opacity": 0
    }
  });

  map.addLayer({
    id: "messaure_outline",
    type: "line",
    source: "messaure",
    "source-layer": "messaure",
    paint: {
      "line-color": "#0A88CF",
      "line-opacity": 0,
      "line-width": 1
    }
  });

  // ─────────────────────────────
  // PINS
  // ─────────────────────────────
  map.addSource("messaure_pin", {
    type: "geojson",
    data: {
      type: "FeatureCollection",
      features: [{
        type: "Feature",
        properties: { name: "Messaure Dam" },
        geometry: {
          type: "Point",
          coordinates: [20.3387, 66.6872]
        }
      }]
    }
  });

  map.addSource("town_pin", {
    type: "geojson",
    data: {
      type: "FeatureCollection",
      features: [{
        type: "Feature",
        properties: { name: "Njuoravuolle" },
        geometry: {
          type: "Point",
          coordinates: [20.15, 66.75]
        }
      }]
    }
  });

  map.addLayer({
    id: "messaure_pin_circle",
    type: "circle",
    source: "messaure_pin",
    paint: {
      "circle-radius": 7,
      "circle-color": water_color,
      "circle-stroke-color": "white",
      "circle-stroke-width": 2
    }
  });

  map.addLayer({
    id: "town_pin_circle",
    type: "circle",
    source: "town_pin",
    paint: {
      "circle-radius": 7,
      "circle-color": town_color,
      "circle-stroke-color": "white",
      "circle-stroke-width": 2
    }
  });

  // ─────────────────────────────
  // CAMERA
  // ─────────────────────────────
  map.fitBounds(
    [19.717, 66.5988, 20.4937, 66.8302],
    { padding: 40, maxZoom: 15 }
  );
});

// ─────────────────────────────
// 2-AXIS CONTROLS
// ─────────────────────────────

// top slider (terrain)
terrain.addEventListener("input", () => {
  map.setPaintProperty(
    "historical-map",
    "raster-opacity",
    0.7 * terrain.value / 100
  );
});

// right slider (flood reveal)
flood.addEventListener("input", () => {
  const v = flood.value / 100;

  map.setPaintProperty(
    "messaure_fill",
    "fill-opacity",
    Math.pow(v, 2.2)
  );

  map.setPaintProperty(
    "messaure_outline",
    "line-opacity",
    v
  );
});
</script>
